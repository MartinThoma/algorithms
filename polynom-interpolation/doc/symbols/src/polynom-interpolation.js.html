 21 /** 
 22  * Calculates coordinates from worldspace to screenspace
 23  * @param {Number} x the coordinate you want to transform
 24  * @param {bool} isX true iff x is a x-coordinate, otherwise false
 25  * @return {Number} transformed coordinate
 26  */
 27 function c(x, isX) {
 28     if (isX) {
 29         return STRETCH_X * (x + X_OFFSET);
 30     }
 31 
 32     return STRETCH_Y * (-x + Y_OFFSET);
 33 }

 35 /** 
 36  * Calculates coordinates from screenspace to worldspace
 37  * @param {Number} x the coordinate you want to transform
 38  * @param {bool} isX true iff x is a x-coordinate, otherwise false
 39  * @return {Number} transformed coordinate
 40  */
 41 function r(x, isX) {
 42     if (isX) {
 43         return x / STRETCH_X - X_OFFSET;
 44     }
 45 
 46     return -x / STRETCH_Y + Y_OFFSET;
 47 }

233 /**
234  * Calculate spline and draw it to canvas.
235  * done like in http://code.activestate.com/recipes/457658-cubic-spline-interpolator/
236  * @param {function as string} f
237  * @param {string} color
238  * @return {undefined}
239  */
240 function spline(f, color) {
241     var x,y;
242     var points = drawEquallySpacedPoints(f, color);
243 
244     f = 'y=' + f;
245     for(var i=0; i<points.length-1; i++) {
246         var u = points[i].x;
247         x = u;
248         eval(f);
249         var FU = y;
250         var j=i+1;
251 
252         var v = points[j].x;
253         x = v;
254         eval(f);
255         var FV = y;
256 
257         // I don't have a derivate :-/
258         var h = 0.01;
259         x = u+h;
260         eval(f);
261         var DU = (y - FU) / h;
262         x = v + h;
263         eval(f);
264         var DV = (y - FV) / h;
265 
266         var denom = Math.pow((u - v),3);
267         var A = ((-DV - DU) * v + (DV + DU) * u + 2 * FV - 2 * FU) / denom;
268         var B = -((-DV - 2 * DU) * Math.pow(v,2)  + u * ((DU - DV) * v + 3 * FV - 3 * FU) + 3 * FV * v - 3 * FU * v + (2 * DV + DU) * Math.pow(u,2)) / denom;
269         var C = (- DU * Math.pow(v,3)  + u * ((- 2 * DV - DU) * Math.pow(v,2)  + 6 * FV * v  - 6 * FU * v) + (DV + 2 * DU) * Math.pow(u,2) * v + DV * Math.pow(u,3)) / denom;
270         var D = -(u *(-DU * Math.pow(v,3)  - 3 * FU * Math.pow(v,2)) + FU * Math.pow(v,3) + Math.pow(u,2) * ((DU - DV) * Math.pow(v,2) + 3 * FV * v) +  Math.pow(u,3) * (DV * v - FV)) / denom;
271         //var m = {'a': A, 'b': B, 'c': C, 'd': D, 'u':u, 'v':v};
272 
273         context.strokeStyle = color;
274         context.beginPath();
275         var isFirst = true;
276         for (var x = u; x <= v; x += 0.01) {
277             y = A*x*x*x + B*x*x + C*x + D;
278             if (isFirst) {
279                 context.moveTo(c(x, true), c(y, false));
280                 isFirst = false;
281             } else {
282                 context.lineTo(c(x, true), c(y, false));
283             }
284         }
285         context.stroke();
286     }
287     
288 }

290 /**
291  * Calculated positions of equally spaced points and draw them to
292  * canvas.
293  * @param {function as string} f
294  * @param {string} color
295  * @return {Array} list of equally spaced points as list of maps

297 function drawEquallySpacedPoints(f, color) {
298     var context = canvas.getContext('2d');
299     context.beginPath();
300     context.strokeStyle = color;
301     f = 'y=' + f;
302     var n = parseInt(document.getElementById("N_EVALUATION_POINTS").value, 10);

304     var X_TO = parseFloat(document.getElementById("X_TO").value, 10);
305     var evaluationSteps = (X_TO - X_FROM) / n;
306     var pointList = new Array(n + 1);
307     if (evaluationSteps > 0) {
308         var i = 0;
309         var y = 0;
310         for (var x = X_FROM; x <= X_TO; x += evaluationSteps) {
311             eval(f);
312             pointList[i] = {
313                 "x": x,
314                 "y": y
315             };
316             i++;
317         }
318         drawPointsGeneral(pointList, color);
319     }
320     return pointList;
321 }

323 function affineTransformation(y, a, b) {
324     return (0.5 * y) * (b - a); // - (a+b)/(b-a);

383 /**
384  * Generate a linear system of equations and store them to a 
385  * n(n+1) matrix.
386  * @param {Array} points
387  * @return {Array} 
388  */
389 function setGauss(points) {
390     var n = points.length - 1;
391     var A = new Array(n + 1);
392     for (var i = 0; i < n + 1; i++) {
393         A[i] = new Array(n + 2);
394         if (points[i] != undefined) {
395             var x = points[i].x;
396             for (var j = 0; j < n + 1; j++) {
397                 A[i][j] = Math.pow(x, j);
398             }
399             A[i][n + 1] = points[i].y;
400         }
401     }
402     return A;
403 }

405 /** 
406  * Solve a linear system of equations given by a nn matrix A 
407  * with a n1 result vector b. 
408  * @param {matrix} A|b
409  * @return {array} x
410  */
411 function gauss(A) {
412     var n = A.length;
413 
414     for (var i = 0; i < n; i++) {
415         // Search for maximum in this column
416         var maxEl = Math.abs(A[i][i]);
417         var maxRow = i;
418         for (var k = i + 1; k < n; k++) {
419             if (Math.abs(A[k][i]) > maxEl) {
420                 maxEl = Math.abs(A[k][i]);
421                 maxRow = k;
422             }
423         }
424 
425         // Swap maximum row with current row (column by column)
426         for (var k = i; k < n + 1; k++) {
427             var tmp = A[maxRow][k];
428             A[maxRow][k] = A[i][k];
429             A[i][k] = tmp;
430         }
431 
432         // Make all rows below this one 0 in current column
433         for (k = i + 1; k < n; k++) {
434             var c = -A[k][i] / A[i][i];
435             for (var j = i; j < n + 1; j++) {
436                 if (i == j) {
437                     A[k][j] = 0;
438                 } else {
439                     A[k][j] += c * A[i][j];
440                 }
441             }
442         }
443     }
444 
445     // Solve equation Ax=b for an upper triangular matrix A
446     var x = new Array(n);
447     for (var i = n - 1; i > -1; i--) {
448         x[i] = A[i][n] / A[i][i];
449         for (var k = i - 1; k > -1; k--) {
450             A[k][n] -= A[k][i] * x[i];
451         }
452     }
453     return x;
454 }

492 /**
493  * Draw polynomial and write it to the document.
494  * @param {DOMElement} canvas
495  * @param {Array} polynomial
496  * @param {string} color
497  * @return {undefined}
498  */
499 function drawPolynomial(canvas, polynomial, color) {

501     context.beginPath();
502     context.strokeStyle = color;
503     var evaluationSteps = parseFloat(document.getElementById("evaluationSteps").value);

506             var y = 0;
507             for (var power in polynomial) {
508                 y += polynomial[power] * Math.pow(x, power);
509             }
510 
511             if(c(y,false) > canvas.height) {
512                 y=r(canvas.height, false);
513             }
514 
515             if(c(y,false) < 0) {
516                 y=r(0, false);
517             }
518 
519 
520             if (x == X_MIN) {
521                 context.moveTo(c(x, true), c(y, false));
522             } else {
523                 context.lineTo(c(x, true), c(y, false));
524             }
525         }
526         context.stroke();
527     }
528 
529     // Write polynomial to document
530     var mathEl = document.getElementById("polynomial");
531     mathEl.innerHTML = '$p(x) = ';
532     for (var power in polynomial) {
533         if (polynomial[power] !== 0) {
534             if (polynomial[power] == 1) {
535                 mathEl.innerHTML += '+x^' + power + ' ';
536             } else {
537                 mathEl.innerHTML += '+' + polynomial[power] + 'x^' + power + ' ';
538             }
539         }
540     }
541     mathEl.innerHTML += '$';
542 }

544 /**
545  * Calcualte euklidean distance of two points.
546  * @param {Map} p1
547  * @param {Map} p2
548  * @return {Number} euklidean distance of p1 and p2
549  */
550 function euklideanDist(p1, p2) {
551     return Math.sqrt(
552         Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
553 }

555 /** 
556  * Add or remove a point.
557  * @param {event} event
558  * @param {DOMElement} canvas
559  * @param {map} mouseCoords
560  * @return {undefined}
561  */
562 function addPoint(event, canvas, mouseCoords) {
563     if (event.ctrlKey) {
564         // remove point that is nearest to mouse coords
565         if (points.length >= 1) {
566             var nearestMouseCIndex = 0;
567             var nearestDist = euklideanDist(mouseCoords, points[0]);
568             var mCoords = {
569                 "x": r(mouseCoords.x, true),
570                 "y": r(mouseCoords.y, false)
571             };
572             for (var i = 1; i < points.length; i++) {
573                 var tmpDist = euklideanDist(mCoords, points[i]);
574                 if (tmpDist < nearestDist) {
575                     nearestDist = tmpDist;
576                     nearestMouseCIndex = i;
577                 }
578             }
579             points.splice(nearestMouseCIndex, 1);
580         }
581     } else {
582         points.push({
583             "x": r(mouseCoords.x, true),
584             "y": r(mouseCoords.y, false)
585         });
586     }
587 }
